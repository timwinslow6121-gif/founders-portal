import phonenumbers

def normalize_e164(raw_phone: str, default_region: str = "US") -> str | None:
    try:
        parsed = phonenumbers.parse(raw_phone, default_region)
        if phonenumbers.is_valid_number(parsed):
            return phonenumbers.format_number(parsed, phonenumbers.PhoneNumberFormat.E164)
    except phonenumbers.NumberParseException:
        pass
    return None
```

    app/comms/__init__.py

    Communications Hub blueprint — Dialpad webhooks, Calendly, Google Meet, SMS templates.
    """
    from flask import Blueprint, jsonify
    from flask_login import current_user
    from app.extensions import db

    comms_bp = Blueprint('comms', __name__, url_prefix='/comms')


    @comms_bp.app_context_processor
    def inject_unmatched_count():
        """Inject unresolved unmatched-call count into all templates for sidebar badge."""
        from app.models import UnmatchedCall
        try:
            if current_user.is_authenticated:
                agency_id = getattr(current_user, 'agency_id', None)
                if agency_id:
                    count = UnmatchedCall.query.filter_by(
                        agency_id=agency_id,
                        resolved=False
                    ).count()
                    return {"unmatched_call_count": count}
        except Exception:
            pass
        return {"unmatched_call_count": 0}


    @comms_bp.route("/health")
    def health():
        return jsonify({"status": "ok"}), 200
    ```

    verify_dialpad_webhook:
    - Reads raw POST body as text (the entire body IS the JWT token)
    - Decodes using jwt.decode(token, secret.encode(), algorithms=["HS256"])
    - Returns payload dict on success
    - Calls abort(403) on jwt.InvalidTokenError
    - Secret comes from current_app.config['DIALPAD_HMAC_SECRET']

    verify_calendly_webhook:
    - Reads 'Calendly-Webhook-Signature' header; format: "t=TIMESTAMP,v1=SIGNATURE"
    - Signs: f"{timestamp}.{body}" with HMAC-SHA256 using CALENDLY_WEBHOOK_SECRET
    - Calls abort(403) if signature mismatch OR event is >300 seconds old (replay protection)
    - NOTE: uses hmac.new (stdlib), not PyJWT

    verify_retell_webhook:
    - Reads 'x-retell-signature' header
    - HMAC-SHA256 of raw body bytes using RETELL_WEBHOOK_SECRET (which IS the Retell API key)
    - Base64-encodes computed digest and compares with provided sig
    - Calls abort(403) on mismatch
    - NOTE: confidence on base64 step is LOW — add a comment: "# Verify against Retell SDK source if signature fails"
