    **Create app/comms/resolution.py with 2 routes:**

    Route 1 — GET /comms/resolution:
    ```python
    @comms_bp.route("/resolution")
    @login_required
    def resolution_queue():
        q = UnmatchedCall.query.filter_by(agency_id=current_user.agency_id, resolved=False)
        if not current_user.is_admin:
            q = q.filter_by(agent_id=current_user.id)
        calls = q.order_by(UnmatchedCall.created_at.desc()).all()
        return render_template("comms/unmatched_calls.html", calls=calls)
    ```

    Route 2 — POST /comms/resolution/<int:call_id>/link:
    ```python
    @comms_bp.route("/resolution/<int:call_id>/link", methods=["POST"])
    @login_required
    def link_unmatched_call(call_id):
        uc = UnmatchedCall.query.filter_by(
            id=call_id, agency_id=current_user.agency_id, resolved=False
        ).first_or_404()
        customer_id = request.form.get("customer_id", type=int)
        customer = Customer.query.filter_by(id=customer_id, agency_id=... ).first_or_404()
        # NOTE: agency_id not on Customer yet (Plan 07) — omit agency_id filter for now
        note = CustomerNote(
            customer_id=customer.id,
            agency_id=current_user.agency_id,
            agent_id=current_user.id,
            note_type="call" if uc.provider != "calendly" else "appointment_scheduled",
            note_text=f"Linked from unmatched call — {uc.from_number}",
            dialpad_call_id=uc.call_sid,
            contact_method="phone",
        )
        db.session.add(note)
        uc.resolved = True
        uc.resolved_at = datetime.utcnow()
        uc.resolved_by_id = current_user.id
        uc.resolved_note_id = note.id
        db.session.commit()
        flash(f"Call linked to {customer.display_name}", "success")
        return redirect(url_for("comms.resolution_queue"))
    ```
